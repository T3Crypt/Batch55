SEQUELIZE :

Setting awal :
# npm install --save-dev sequelize-cli
# npx sequelize-cli init

Add Type :
# npx sequelize-cli model:generate --name type_tb --attributes name:string

Add Hero :
# npx sequelize-cli model:generate --name heroes_tb --attributes name:string,type_id:integer,photo:string

Migrasi / Buat Table :
# npx sequelize-cli db:migrate

Masuk ke db ml_project : 
#  psql -U postgres -h 127.0.0.1 -d ml_project

Liat seluruh isinya :
SELECT * FROM "heroes_tb";

Tampilkan seluruh data dari table heroes beserta type dari hero tsb :
SELECT heroes_tb.id, heroes_tb.name, type_tb.name as type_name, heroes_tb.photo, heroes_tb."createdAt", heroes_tb."updatedAt"
FROM heroes_tb
JOIN type_tb ON heroes_tb.type_id = type_tb.id;

Tampilkan seluruh data heroes berdasarkan type tertentu (Poke) :
SELECT heroes_tb.id, heroes_tb.name, type_tb.name as type_name, heroes_tb.photo, heroes_tb."createdAt", heroes_tb."updatedAt"
FROM heroes_tb
JOIN type_tb ON heroes_tb.type_id = type_tb.id
WHERE type_tb.name = 'Poke';

Tampilkan spesifik data heroes(Alpha) dengan type(Damage) nya :
SELECT heroes_tb.id, heroes_tb.name, type_tb.name as type_name, heroes_tb.photo, heroes_tb."createdAt", heroes_tb."updatedAt"
FROM heroes_tb
JOIN type_tb ON heroes_tb.type_id = type_tb.id
WHERE heroes_tb.name = 'Alpha' AND type_tb.name = 'Damage';
